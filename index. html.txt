<script>
    var data = {
        'gt': '<b>Giao thong:</b> Xem cac tuyen duong chinh.',
        'dh': '<b>Dia hinh:</b> Tim cac vung cao (DEM2) de tranh ngap.',
        'c1': '<b>Cap 1:</b> Nuoc bat dau dang o vung ven song Ba.',
        'c2': '<b>Cap 2:</b> Nguy hiem, nhieu tuyen duong bi chia cat.',
        'c3': '<b>Cap 3:</b> Ngap lut nghiem trong toan huyen.'
    };

    function show(id, btn) {
        // An het anh
        var imgs = document.getElementsByClassName('map-layer');
        for (var i = 0; i < imgs.length; i++) {
            imgs[i].style.display = 'none';
        }
        // Hien anh duoc chon
        document.getElementById(id).style.display = 'block';

        // Cap nhat chu
        document.getElementById('msg').innerHTML = data[id];

        // Doi mau nut
        var btns = document.getElementsByTagName('button');
        for (var j = 0; j < btns.length; j++) {
            btns[j].classList.remove('active');
        }
        btn.classList.add('active');
    }
</script>
